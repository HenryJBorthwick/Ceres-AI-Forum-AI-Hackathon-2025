% Your Big Idea (2 minutes) - Slide 1: Overview
\begin{frame}{Our Innovation: Ceres Platform}
  Ceres uses Google DeepMind's Alpha Earth geospatial embeddings combined with LightGBM ML models to predict IPC phases.
  \vspace{1em}
  \begin{itemize}
    \item Predicts population in IPC phases 1-5 for 2025-2026
    \item Interactive web interface for easy analysis
    \item Meets UN brief by addressing SDG 2 with AI
  \end{itemize}
\end{frame}

% Your Big Idea - Slide 2: Technology
\begin{frame}{How It Works: Technical Overview}
  \begin{columns}
    \column{0.5\textwidth}
      \textbf{Backend (FastAPI + LightGBM):}
      \begin{itemize}
        \item Loads 64-band satellite embeddings
        \item Predicts phase percentages
        \item Aggregates historical IPC data
      \end{itemize}
    \column{0.5\textwidth}
      \textbf{Frontend (React):}
      \begin{itemize}
        \item Country/Region selection
        \item Interactive charts with Recharts
        \item Visualizes predictions vs historical data
      \end{itemize}
  \end{columns}
  \vspace{1em}
  Model Accuracy: 81.1\% phase classification
\end{frame}
