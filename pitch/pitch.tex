% Your Big Idea (2 minutes) - Slide 1: Overview
\begin{frame}{Our Innovation: Ceres Platform}
  Ceres AI IPC Prediction leverages Google DeepMind's Alpha Earth geospatial embeddings combined with LightGBM ML models to predict IPC phases.
  \vspace{1em}
  \begin{itemize}
    \item Embeddings offer rich information about the environment.
    \item In a format AI models love.
    \item Collected and updated continuously.
    \item Versatile for use in a variety metrics.
  \end{itemize}
\end{frame}

% Your Big Idea - Slide 2: Technology
\begin{frame}{How It Works: Technical Overview}
  \begin{center}
    \begin{enumerate}
      \item \textbf{Input Data Sources:}
      \begin{itemize}
        \item Country/Region IPC historical data
        \item Country/Region Satellite embeddings
      \end{itemize}
      \vspace{0.5em}
      \item \textbf{AI Model:} LightGBM trains on the data
      \vspace{0.5em}
      \item \textbf{Backend:} Python FastAPI manages and serves predictions and historical data
      \vspace{0.5em}
      \item \textbf{Frontend:} React + Recharts displays the data in a human friendly way.
    \end{enumerate}
  \end{center}
\end{frame}
